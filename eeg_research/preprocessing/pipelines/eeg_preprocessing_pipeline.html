 50def trackcalls(
 51    func: Callable[ParamType, ReturnType],
 52) -> Callable[ParamType, ReturnType]:
 53    """Decorator to track if a method have been called.
 54
 55    Args:
 56        func (Callable[ParamType, ReturnType]): The function to decorate
 57
 58    Returns:
 59        Callable[ParamType, ReturnType]: The function decorated
 60    """
 61
 62    @functools.wraps(func)
 63    def wrapper(*args: ParamType.args, **kwargs: ParamType.kwargs) -> ReturnType:
 64        setattr(wrapper, "has_been_called", True)
 65        return func(*args, **kwargs)
 66
 67    setattr(wrapper, "has_been_called", False)
 68    return wrapper

 85class EEGpreprocessing:
 86    """Class that wrap several preprocessing techniques.
 87
 88    This class is a wrapper around the pyprep and asrpy pipelines and other
 89    preprocessing steps.
 90    """
 91
 92    def __init__(self, eeg_filename: str | os.PathLike) -> None:
 93        """Constructor for the EEGpreprocessing object.
 94
 95        Args:
 96            eeg_filename (str or os.PathLike): the path to the EEG file
 97        """
 98        self.eeg_filename = eeg_filename
 99        self.raw = mne.io.read_raw(eeg_filename, preload=True)
100        channels_map = utils.map_channel_type(self.raw)
101        self.raw = utils.set_channel_types(self.raw, channels_map)
102
103    def set_annotations_to_raw(
104        self, events_filename: str | os.PathLike
105    ) -> "EEGpreprocessing":
106        """Automatically set the annotations on the raw object.
107
108        It takes care of the subtelties of the CST dataset. It handles correctly
109        the timestamps and the timezone set.
110
111        Args:
112            events_filename (str of os.PathLike): the path to the events file
113                                                  saved in a tsv or csv format
114
115        Returns:
116            self
117        """
118        self.events = pd.read_csv(events_filename)
119        events_renamed = self.events.copy()
120        events_renamed.loc[
121            events_renamed["StimMarkers_alpha"].str.contains("Crash"),
122            "StimMarkers_alpha",
123        ] = "Crash"
124
125        timestamp = [
126            datetime.datetime.fromtimestamp(t) for t in events_renamed["timestamps"]
127        ]
128
129        events_renamed["timestamps"] = [
130            t.replace(tzinfo=datetime.timezone.utc) for t in timestamp
131        ]
132
133        description = events_renamed["StimMarkers_alpha"].values
134
135        onsets = [
136            t.seconds for t in events_renamed["timestamps"] - self.raw.info["meas_date"]
137        ]
138
139        self.annotations = self.raw.annotations.append(
140            onset=onsets, duration=np.zeros((len(onsets))), description=description
141        )
142
143        self.raw.set_annotations(self.annotations)
144        return self
145
146    @trackcalls
147    def set_montage(self, montage: str = "easycap-M1") -> "EEGpreprocessing":
148        """Wrapper around mne.channels.make_standard_montage('easycap-M1').
149
150        The montage is hardcoded to 'easycap-M1' because it is the one used in
151        the CST dataset.
152
153        Args:
154            montage (str): The montage name. Has to be one among the known
155                           standard montage in MNE. It is possible to get all
156                           possible values by running:
157                           `mne.channels.get_builtin_montages()`
158
159        Returns:
160             EEGpreprocessing object
161        """
162        self.montage = mne.channels.make_standard_montage(montage)
163        self.raw.set_montage(self.montage)
164        return self
165
166    def run_prep(self) -> "EEGpreprocessing":
167        """Run the pyprep pipeline on the raw object.
168
169        Returns:
170            EEGpreprocessing object
171        """
172        if self.set_montage.has_been_called:  # type: ignore[attr-defined]
173            prep_params = {
174                "ref_chs": "eeg",
175                "reref_chs": "eeg",
176                "line_freqs": np.arange(60, self.raw.info["sfreq"] / 2, 60),
177            }
178            prep_obj = prep.PrepPipeline(
179                self.raw, montage=self.montage, prep_params=prep_params
180            )
181            prep_obj.fit()
182            self.raw = prep_obj.raw_eeg
183            return self
184
185        else:
186            raise MissingStepError(
187                "You must set a montage before. Please run `set_montage`"
188            )
189
190    def run_asr(self) -> "EEGpreprocessing":
191        """Run the asrpy pipeline on the raw object.
192
193        Returns:
194            EEGpreprocessing object
195        """
196        if self.set_montage.has_been_called:  # type: ignore[attr-defined]
197            asr_obj = asr.ASR(sfreq=self.raw.info["sfreq"], cutoff=10)
198            asr_obj.fit(self.raw)
199            self.raw = asr_obj.transform(self.raw)
200            return self
201        else:
202            raise MissingStepError(
203                "You must set a montage before. Please run `set_montage`"
204            )
205
206    def save(self, filename: str | os.PathLike) -> "EEGpreprocessing":
207        """Save the data.
208
209        Args:
210            filename: the name of the file to save
211        """
212        mne.export.export_raw(filename, self.raw)
213        return self

216def main(
217    reading_filename: str | os.PathLike,
218    saving_filename: str | os.PathLike,
219    methods: list[str],
220) -> None:
221    """Wrapper to access to the preprocess from CLI.
222
223    Args:
224        reading_filename (str | PathLike): The EEG filename (full path)
225        saving_filename (str | PathLike): Where to save the preprocessed data
226        methods (list of str): The methods to use in the pipeline
227
228    Return:
229        None
230    """
231    preprocess = EEGpreprocessing(reading_filename)
232    existing_method = [
233        cl_method for cl_method in dir(preprocess) if not cl_method.startswith("__")
234    ]
235    methods.insert(0, "set_montage")
236    for method in methods:
237        if method in existing_method:
238            getattr(preprocess, method)()
239        else:
240            raise AttributeError(f"""`
241Wrong input method name. Should be one of the following:
242{"\n".join(existing_method)}
243""")
244    preprocess.save(saving_filename)
245
246
247if __name__ == "__main__":
248    parser = argparse.ArgumentParser(
249        prog="eeg_preprocessing_pipeline",
250        description="""
251        This code uses different preprocessing methods/techniques to prepare EEG
252        data.
253                        """,
254        epilog="""
255        The preprocessing methods has to be sepcified by by calling --methods
256        when calling the script. It can be several methods, names have to be
257        separated by a comma. Beware the order matters
258        """,
259    )
260
261    parser.add_argument(
262        "reading_filename",
263        default=None,
264        help="The full path of the EEG file to process",
265    )
266    parser.add_argument(
267        "saving_filename",
268        default=None,
269        help="The full path where to save the processed EEG file",
270    )
271    parser.add_argument("--methods", nargs="*")
272    kwargs_namespace = parser.parse_args()
273    main(**kwargs_namespace.__dict__)
